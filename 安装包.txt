# OpenCV (opencv-python 和 opencv-contrib-python):
# - 提供图像处理、视频分析、目标检测与跟踪等功能的开源计算机视觉库。

# 安装 OpenCV 包
pip install opencv-python
# 引用 OpenCV 包
import cv2

# 安装 OpenCV 扩展包
pip install opencv-contrib-python

# 安装 CMake:
# - 跨平台的构建系统，用于管理复杂的项目结构和依赖关系。
pip install cmake

# 安装 scikit-image:
# - 基于 Python 的图像处理库，提供滤波、边缘检测、形态学操作等功能。
pip install scikit-image
# 引用 scikit-image 包
import skimage

# 安装 dlib:
# - 提供机器学习、图像处理和计算机视觉相关功能的 C++ 库，广泛应用于人脸检测、识别等领域。
pip install dlib
# 引用 dlib 包
import dlib

# 安装 face_recognition:
# - 基于 dlib 的人脸识别库，用于识别图像中的人脸并进行特征比对。
pip install face_recognition
# 引用 face_recognition 包
import face_recognition

# 安装 requests:
# - 发送 HTTP 请求和处理响应的 Python HTTP 库，用于与 Web 服务进行通信。
pip install requests
# 引用 requests 包
import requests

# 安装 matplotlib:
# - 生成各种类型图表和图形的 Python 2D 绘图库，包括线图、散点图、直方图等。
pip install matplotlib
# 引用 matplotlib 包
import matplotlib.pyplot as plt

# 安装 Jupyter:
# - 交互式计算环境，支持多种编程语言，方便进行数据分析、可视化和演示。
pip install jupyter

# 安装 scikit-learn:
# - 提供各种机器学习算法和工具的机器学习库，包括分类、回归、聚类、降维等功能。
pip install scikit-learn
# 引用 scikit-learn 包
import sklearn

# CVZone:
# - 用于计算机视觉和图像处理的 Python 库，提供目标检测、姿态估计、手势识别等功能。
# 安装 CVZone
pip install cvzone
# 引用 CVZone
import cvzone

# Mediapipe:

# 安装 Mediapipe
pip install mediapipe
